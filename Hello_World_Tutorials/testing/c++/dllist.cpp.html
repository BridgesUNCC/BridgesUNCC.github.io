DLelement<StudentInfo> *insertFront(DLelement<StudentInfo> *front,
	DLelement<StudentInfo> *new_el) {
	if (front == nullptr)
		return new_el;

	new_el->setNext(front);
	front->setPrev(new_el);

	return new_el;
}
