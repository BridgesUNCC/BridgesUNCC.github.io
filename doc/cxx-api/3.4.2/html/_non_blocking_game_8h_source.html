206  void handleFrameRate() {
207  using std::chrono::seconds;
208  using std::chrono::microseconds;
209  using std::chrono::duration_cast;
210 
211  microseconds frametime = duration_cast<microseconds>(seconds(1l)) / (int)fps;
212 
213  localclock::time_point theoretical_next_frame = timeOfLastFrame + frametime;
214 
215  auto wait_time = theoretical_next_frame - localclock::now();
216 
217  if (wait_time.count() > 0) {
218  microseconds wait_time_in_us = duration_cast<microseconds>(wait_time);
219  usleep(wait_time_in_us.count());
220  }
221
