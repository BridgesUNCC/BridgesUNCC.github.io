352  void setValue(string name) {
353  //marking static to avoid rebuilding the regexp every time.
354  //greedy, so checks for 8(RRGGBBAA), then 6(RRGGBB) then 4(RGBA) then 3(RGB)
355  static const regex HEX_RANGE("^#([[:xdigit:]]{8}|[[:xdigit:]]{6}|[[:xdigit:]]{3,4})$");
356 
357  for (char& c : name) {
358  c = (char) tolower((unsigned char)c); //convert to lowercase
359  }
360  auto it = ColorNames().find(name);
361 
362  if (it != ColorNames().end()) {
363  channels = it->second; //Named value
364  }
365  else if (regex_match(name, HEX_RANGE)) { //#Hex value
366  name.erase(0, 1); //removes "#"
367  // alpha value, overwritten if present by loop number
368  // of chars representing a channel
369  channels.at(3) = 255;
370  const int chanChars = (name.size() == 3 || name.size() == 4) ? 1 : 2;
371  //unit place scale factor, handles channel size variance
372  const int chanMultiplier = (chanChars == 1) ? 17 : 1;
373  for (size_t i = 0; i < name.size() / chanChars; i++) {
374  //converts and save hex val to rgba val
375  channels.at(i) = (int) strtol(name.substr(i * chanChars,
376  chanChars).c_str(), nullptr, 16) * chanMultiplier;
377  }
378  }
379  else { //invalid color
380  string errStr = "Invalid Color: " + name + "\nMust be a hexadecimal(#RRGGBBAA, #RRGGBB, #RGBA, or #RGB) color representation\nOr one of these supported named colors: ";
381  for (const auto& p : ColorNames()) {
382  errStr += " \"" + p.first + "\"";
383  }
384  errStr += "\n";
385  throw errStr;
386  }
