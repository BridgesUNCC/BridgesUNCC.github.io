940  int dimx = dimensions[0].GetInt();

956  if (debug())
957  std::cerr << "decoding RAW" << std::endl;
958  if (debug())
959  std::cerr << "length: " << decoded.size() << std::endl;
960  if (decoded.size() < dimx * dimy * 4)
961  throw "Malformed ColorGrid JSON: nodes is smaller than expected";
962

972  bridges::Color c ((int)decoded[base],
973  (int)decoded[base + 1],
974  (int)decoded[base + 2],
975  (int)decoded[base + 3]
976  );
977

988  int currentInCG = 0;

991  throw "Malformed ColorGrid JSON: nodes is not a multiple of 5";
992 
993 
994

1011  int posY = currentInCG % dimy;
1012  if (posX >= dimx || posY >= dimy) {

1017  cg.set(posX, posY, c);
1018 
1019  currentInCG++;
1020  repeat --;
1021  }
1022  }
1023  if (debug())
1024  std::cerr << "written " << currentInCG << " pixels" << std::endl;
1025  if (currentInCG != dimx * dimy)
1026  throw "Malformed ColorGrid JSON: Not enough data in nodes";
1027  }
1028 
1029  return cg;
1030  }
1031  catch (rapidjson_exception re) {
