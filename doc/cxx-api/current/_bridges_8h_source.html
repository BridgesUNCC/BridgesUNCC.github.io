356  void visualize() {

365  if (ds == "Array")

367  else if (ds == "llist")
368  visualize_LinkedList(url_to_server, url_to_vis);

370  visualize_DoublyLinkedList(url_to_server, url_to_vis);
371  else if (ds == "tree")
372  visualize_BinaryTree(url_to_server, url_to_vis);
