class AudioTest(AudioClip):

    def generate_sine(self, bit_depth: int) -> None:
        for s in range(self.sample_count):
            for c in range(self.num_channels):
                val = math.sin(s / 20)
                self.set_sample(c, s, val * ((2 ** bit_depth / 2.0) - 1.0))
